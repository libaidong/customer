305

308

310 
311  LOG_INFO("IO CFG, IO CNT, %lu, DEV CNT, %lu\r\n", IOCount, DeviceCount);
312 
313  /* Check if IO count is not zero */
314  if(!IOCount)

319  }
320 
321  /* Get device info reference ptr. */

325  DeviceInfo = realloc(DeviceInfo, (sizeof(DEVICE_INFO) * (DeviceCount + IOCount)));

329  return -1;
