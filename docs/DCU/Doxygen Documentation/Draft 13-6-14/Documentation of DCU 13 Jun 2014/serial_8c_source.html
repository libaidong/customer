212  * CRTSCTS : Hardware flow control (clear to send and request to send).

217  * */

234  {
235  /* Set parity bit. */
236  /* Setting for odd parity.*/
237  if(Parity == PARODD)

239  NewTIO.c_cflag |= PARENB;
240  NewTIO.c_cflag |= Parity;
241  }
242  /* Set for even parity. */
243  else if(Parity == ~PARODD)
244  {
245  NewTIO.c_cflag |= PARENB;
246  NewTIO.c_cflag &= Parity;
247  }
248  /* Set none parity. */
249  else

252  }
253 
254

258  {
259

261 
262  }
263  /* Set 1 stop bit. */
264  else
265  {
266  NewTIO.c_cflag &= StopBits;
267  }
268 
269  /* Configure c flag parameters. */
270  NewTIO.c_cflag = NewTIO.c_cflag | BaudRate | DataBits | CLOCAL | CREAD ;
