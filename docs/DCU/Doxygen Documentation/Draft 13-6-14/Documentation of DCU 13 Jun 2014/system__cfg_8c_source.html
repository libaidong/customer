135 int ReadSystemConfig()
136 {
137  unsigned char FileName[100] = {0};
138  unsigned char Tag[LEN_OF_INTERFACE_TAG] = "ETH";
139  char RetVal = -1;
140  ETH_INTERFACE Eth;
141  SERIAL_INTERFACE Serial;
142  unsigned char Iterator = 0;
143 
144  /* Get file path and name of system config */
145  GET_SYSTEM_CONFIG_FILE_PATH(FileName);
146 
147  /* Clear the system config structure */
148  memset(&SystemConfig, '\0', sizeof(SystemConfig));
149 
150  LOG_INFO("XML, CHK, %s, FILE\r\n", FileName);
151 
152  /* Check if config file exists */
153  if(0 == CheckIfPathExists(FileName, NULL))
154  {
155  /* Parse the xml file using the parser and fill all the details in SYSTEM_CFG_STRUCT variable */
156  if(0 == ParseXMLFile(FileName, (unsigned char *)&SystemConfig, SystemConfigFileTag, XML_MAX_SYSTEM_CONFIG_FILE_TAG))
157  {
158  RetVal = 0;
159 
160  /* Print the system config parsed from gthe xml file. */
161  PrintSystemCfg(&SystemConfig);
162  }
163  else
164  {
165  /* Add event in the event file */
166  AddEvent("SYS CFG FILE PARSE ERR", MACAddress, NULL);
167  }
168  }
169  else
170  {
171  /* Add event in the event file */
172  AddEvent("SYS CFG FILE MISSING", MACAddress, NULL);
173  }
174 
175  /* Get the ETH interface name if any from the system. */
176  GetEthInterface(&Eth);
177 
178  /* Copy the interface in the actual system config */
179  strcpy(SystemConfig.EthInterface.InterfaceName, Eth.InterfaceName);
180 
181  /* TODO: Check if config file has IP address. If YES set that and go ahead or else copy the returned IP in the system config structure */
182  LOG_INFO("SYS CFG, IF NAME, %s, IP, %s, GIP, %s, CHMSK, %s\r\n", Eth.InterfaceName, Eth.IP, Eth.GatewayIP, Eth.SubnetMask);
183 
184  /* Register the default interface which is ethernet */
185  RegisterInterface(Tag, INTERFACE_TYPE_ETH, &Eth);
186 
187  LOG_INFO("SYS CFG, IF CNT, %u\r\n", SystemConfig.SerialInterfaceCnt);
188 
189  /* Register serial interafce. */
190  for(Iterator = 0; Iterator < SystemConfig.SerialInterfaceCnt; Iterator++)
191  {
192  Serial = SystemConfig.SerialInterface[Iterator];
193 
194  /* Don't register the serial interface it is disabled */
195  if(Serial.Enable != TRUE)
196  {
197  LOG_ERR(RED"SYS CFG, Serial INF \"%s\" disabled.\r\n"NOCOLOR, Serial.Tag);
198  continue;
199  }
200 
201  /* Register the interface for serial. */
202  RegisterInterface(SystemConfig.SerialInterface[Iterator].Tag, INTERFACE_TYPE_SERIAL, &Serial);
203  }
204 
205  return RetVal;
206 }
