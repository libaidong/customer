244 void *ExecutePolling()
245 {
246  int Iterator;
247  unsigned char DataReqRetry = 0;
248  unsigned long WaitTimerForDataResp = 0;
250  /*Initialize the states and poll timer for all available meters. */
251  for(Iterator = 0; Iterator < MeterCount; Iterator++)
252  {
253  /* Update the state for a meter index. */
254  Meters[Iterator].State = METER_POLLING_STATE_WAIT_FOR_CONN;
255 
256  /* Update the default data to ram storage. */
257  memset(Meters[Iterator].StoragePoolRef + 1, 0x00, GetLengthBasedOnDataType(Meters[Iterator].OutputDataType));
258  }
259 
260  /* polling thread execution while loop. */
261  while(1)
262  {
263  /* Iterate to all available meter for polling. */
264  for(Iterator = 0; Iterator < MeterCount; Iterator++)
265  {
266  switch(Meters[Iterator].State)
267  {
268  /* Meter state as idle when no action to take. */
269  case METER_POLLING_STATE_WAIT_FOR_CONN:
270  break;
271 
272  /* Meter state if connection done. */
273  case METER_POLLING_STATE_CONNECTED:
274  /* Update the poll timer for a meters. */
275  Meters[Iterator].PollTimer = 0;
276 
277  /* Update the state to wait for poll timer elapse for the data request over modbus */
278  Meters[Iterator].State = METER_POLLING_STATE_IS_POLL_TIMER_ELAPSE;
279  break;
280 
281  /* Check for timer elapse to req data. */
282  case METER_POLLING_STATE_IS_POLL_TIMER_ELAPSE:
283  /* Check poll timer to elapse to send the data request. */
284  if(IS_TIMER_ELAPSED(Meters[Iterator].PollTimer))
285  {
286  LOG_INFO("MB_TCP, Set next poll timer, %lu\r\n", Meters[Iterator].PollInterval);
287 
288  /* Set the next polling timer to poll. */
289  SET_TIMER(Meters[Iterator].PollTimer, Meters[Iterator].PollInterval);
290 
291  /* Update the meter context state to send the data request. */
292  Meters[Iterator].State = METER_POLLING_STATE_SEND_DATA_REQ;
293  }
294  break;
295 
296  /* Send the data req on MB TCP. */
297  case METER_POLLING_STATE_SEND_DATA_REQ:
298 
299  /* The the MB TCP packet and send. */
300  FillAndSendDataReq(&Meters[Iterator]);
301 
302  /* Update the meter context state to send the data request. */
303  Meters[Iterator].State = METER_POLLING_STATE_WAIT_FOR_DATA_RESP;
304 
305  /* Update the data response timer for retry. */
306  SET_TIMER(WaitTimerForDataResp, MAX_TIMER_TO_WAIT);
307  break;
308 
309  /* wait for the data response and retry. */
310  case METER_POLLING_STATE_WAIT_FOR_DATA_RESP:
311  /* Check to send the data request for retry. */
312  if(IS_TIMER_ELAPSED(WaitTimerForDataResp))
313  {
314  /* Increment the data request retry count. */
315  DataReqRetry++;
316 
317  /* if data request retry exceeded than fill the default response and wait for next timer elapse */
318  if(DataReqRetry > MAX_RETRY_FOR_DATA_REQ)
319  {
320  DataReqRetry = 0;
321 
322  /* Update the meter context state to fill the default response. */
323  Meters[Iterator].State = METER_POLLING_STATE_FILL_DEFAULT_RESP;
324  }
325  else
326  {
327  /* Update the meter context state to send the data request. */
328  Meters[Iterator].State = METER_POLLING_STATE_SEND_DATA_REQ;
329  }
330  }
331  break;
332 
333  /* Fill the default response in ram if all retries are over. */
334  case METER_POLLING_STATE_FILL_DEFAULT_RESP:
335  /* TODO : fill the default data response. */
336 
337  /* Check for next poll timer to elapse. */
338  Meters[Iterator].State = METER_POLLING_STATE_IS_POLL_TIMER_ELAPSE;
339  break;
340 
341  /* State state when data received successfully. */
342  case METER_POLLING_STATE_SUCC_DATA_RECV:
343  /* Update the state for a meter. */
344  Meters[Iterator].State = METER_POLLING_STATE_IS_POLL_TIMER_ELAPSE;
345  break;
346  }
347  }
348

350  sleep(1);
