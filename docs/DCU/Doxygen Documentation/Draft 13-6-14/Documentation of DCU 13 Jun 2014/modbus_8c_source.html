69 {
70 
71  unsigned char uchCRCHi = 0xFF; /* high byte of CRC initialized */
72  unsigned char uchCRCLo = 0xFF; /* low byte of CRC initialized */
73  unsigned char uIndex; /* will index into CRC lookup table */
74 
75  while((PacketLength)--) /* pass through message buffer */
76  {

78  uchCRCHi = uchCRCLo ^ auchCRCHi[uIndex];

578  {
579  /* Register address found do update the flag. */
580  RegFindFlag = 1;
581 
582  /* Read length of the data that to be read from storage pool. */
583  ByteCnt = *(BMSDataCfg[Iterator].StoragePoolRef);
584 
585  /* If configured data type length is not matched with the received byte count then log the error and return. */
586  if(GetLengthBasedOnDataType(BMSDataCfg[Iterator].InfoRefernce->OutputDataType) != ByteCnt)
587  {
588  LOG_ERR("MODBUS, ERR, WRONG BYTE LENGTH FOUND, REG, %d\n", RegAddr );
589 
590  RegFindFlag = -1;
591 
592  break;
593  }
594  else
595  {
596  /* Incrementing the total byte count as we got the data type. */
597  TotalByteCount += ByteCnt;
598 
599  /* Check for register type i.e function code is valid or not. */
600  if(BMSDataCfg[Iterator].RegType != ModbusInfo->RegType)
601  {
602  if((ModbusInfo->RegType != MB_POINT_TYPE_WRITE_COIL) && (ModbusInfo->RegType != MB_POINT_TYPE_WRITE_MULTIPLE_COIL))
603  {
604  LOG_ERR("MODBUS, VALIDATE BMS REG TYPE, INVALID REGISTER TYPE, BMS, %d, MODBUS, %d\r\n", BMSDataCfg[Iterator].RegType, ModbusInfo->RegType);
605  /* Invalid register type i.e function code. */
606  return ERROR_CODE_INVALID_REG_TYPE;
607  }
608  }
609 
610  /* In case only one byte data is found. */
611  if(ByteCnt == 1)
612  {
613  /* Incrementing the Register address. */
614  RegAddr += ByteCnt;
615  }
616  else
617  {
618  /* Incrementing the Register address. */
619  RegAddr += ByteCnt / 2;
620  }
621  }
622  /* Checking if all the bytes are read. */
623  if(TotalByteCount == TotalReadableBytes)
624  {
625  break;
626  }
627  }
628  else
629  {
630  /* Register address not found. */
631  RegFindFlag = -1;
632  }
633  }
634  /* Register address and number of registers requested from BMS are invalid so give error. */
635  if(RegFindFlag == -1)
636  {
637  LOG_ERR("MODBUS, VALIDATE BMS REG ADDR, INVALID BMS REGISTER ADDRESS, %u\r\n", RegAddr);
638  return ERROR_CODE_INVALID_BMS_REG_ADDR;
639  }
640 
641  return 0;

746 char FillBMSModbusResp(MODBUS_REQ_INFO *ModbusInfo, unsigned char *Buffer, unsigned char ModbusType, unsigned char InterfaceIdx)
747 {
748  unsigned int Offset = 0;
749  unsigned short CheckSum = 0;
750  char OffsetIndex = 0;
751  short OffsetToFill = 0;
752  int RetVal = -1;
753  unsigned char Iterator = 0;
754  unsigned short LengthToReduce = 0;
755  unsigned short TempWriteStatus = 0;
756 
757  LOG_INFO("MODBUS PKT, FILL MB PKT, MT, %d, MEDIA, %d\r\n", ModbusType, GetInterfaceType(InterfaceIdx));
758 
759  /* If type of modbus is RTU and interface is ETH then parse the extra header bytes */
760  if((ModbusType == PT_MB_RTU) && (GetInterfaceType(InterfaceIdx) == INTERFACE_TYPE_ETH))
761  {
762  /* Fill transaction id. */
763  WriteShortToPacket(ModbusInfo->TransactionId, &Buffer[Offset], &Offset);
764 
765  /* Fill protocol identifier. Zero for modbus tcp. */
766  WriteShortToPacket(ModbusInfo->ProtocolId, &Buffer[Offset], &Offset);
767 
768  /* Store offset index. */
769  OffsetIndex = Offset;
770 
771  /* Increment the length offset for the data length feild */
772  Offset += 2;
773 
774  /* The length that will be filled in the length field will be the actual length after the length field
775  so taking the count of length which is not required into a variable so that at the time of length filling
776  we can get the exact length of packet */
777  LengthToReduce = Offset;
778  }
779 
780  /* Fill slave address. */
781  Buffer[Offset++] = ModbusInfo->SlaveAddr;
782 
783  /* Fill register type. */
784  Buffer[Offset++] = ModbusInfo->RegType;
785 
786  /* Check for all type of registers like write multiple coils and all that */
787  if(ModbusInfo->RegType == MB_POINT_TYPE_WRITE_COIL)
788  {
789  /* Writing register address into the packet */
790  WriteShortToPacket(ModbusInfo->RegAddr, &Buffer[Offset], &Offset);
791 
792  /* If write status is found as 1 means on then again write the modbus standard packet value into the packet, to be sent to BMS */
793  if(ModbusInfo->WriteStatus == 1)
794  {
795  TempWriteStatus = 65280;
796  }
797 
798  /* Writing the write status into the packet */
799  WriteShortToPacket(TempWriteStatus, &Buffer[Offset], &Offset);
800 
801  RetVal = 0;
802  }
803  else if(ModbusInfo->RegType == MB_POINT_TYPE_WRITE_MULTIPLE_COIL)
804  {
805  /* Writing register address into the packet */
806  WriteShortToPacket(ModbusInfo->RegAddr, &Buffer[Offset], &Offset);
807 
808  /* Writing the number of registers to write */
809  WriteShortToPacket(ModbusInfo->NoOfRegToRead, &Buffer[Offset], &Offset);
810 
811  RetVal = 0;
812  }
813  else
814  {
815  /* Getting here the storage pool data and filling into the sending buffer */
816  RetVal = GetDataFromStoragePool(ModbusInfo, &Buffer[Offset]);
817  }
818 
819 
820  /* Slave address not present in BMS Info so return error flag. */
821  if(RetVal < 0)
822  {
823  LOG_ERR("MODBUS, SLAVE ADDRESS IS INVALID\n");
824  return -1;
825  }
826 
827  if((ModbusType == PT_MB_RTU) && (GetInterfaceType(InterfaceIdx) == INTERFACE_TYPE_ETH))
828  {
829  /* Updating the offset */
830  Offset += RetVal;
831 
832  /* Get the offset to fill at length field. */
833  OffsetToFill = Offset - LengthToReduce;
834 
835  /* Fill length field. Represents number of remaining bytes in the frame. */
836  WriteShortToPacket(OffsetToFill, &Buffer[OffsetIndex], NULL);
837  }
838  else
839  {
840  /* Updating the offset value */
841  Offset += RetVal;
842 
843  /* Calculate CRC. */
844  CheckSum = CalculateCheckSum(Buffer, Offset);
845 
846  /* Writing the checksum into the packet */
847  WriteShortToPacket(CheckSum, &Buffer[Offset], &Offset);
848  }
849 
850 // PrintByteArray(Buffer, Offset);
851 
852  return Offset;
853 }
